fast_forward.encoder

Custom encoders

Custom encoders can easily be implemented. The preferred way to do this is by subclassing fast_forward.encoder.base.Encoder and overriding the _encode method. This allows the new encoder to make use of batch encoding.

Alternatively, one can use the LambdaEncoder class, which wraps a function that encodes a single piece of text. The following example shows how to do this with a Pyserini query encoder:

pyserini_encoder = pyserini.encode.AnceQueryEncoder("castorini/ance-msmarco-passage")
ance_encoder = LambdaEncoder(pyserini_encoder.encode)

Note that this method is usually less efficient, as the texts are encoded one by one rather than in batches.

18class TransformerEncoder(Encoder):
19    """Uses a pre-trained Transformer model for encoding.
20
21    The outputs corresponding to the CLS token from the last hidden layer are used.

24    def __init__(
25        self,
26        model: "str | Path",
27        device: str = "cpu",
28        model_args: "Mapping[str, Any]" = {},
29        tokenizer_args: "Mapping[str, Any]" = {},

31            "padding": True,

33        },
34        normalize: bool = False,
35    ) -> None:
36        """Create a Transformer encoder.
37
38        :param model: Pre-trained Transformer model (name or path).
39        :param device: PyTorch device.
40        :param model_args: Additional arguments for the model.
41        :param tokenizer_args: Additional arguments for the tokenizer.
42        :param tokenizer_call_args: Additional arguments for the tokenizer call.
43        :param normalize: L2-normalize output representations.
44        """
45        super().__init__()
46        self._model = AutoModel.from_pretrained(model, **model_args)
47        self._model.to(device)
48        self._model.eval()
49        self._tokenizer = AutoTokenizer.from_pretrained(model, **tokenizer_args)
50        self._device = device
51        self._tokenizer_call_args = tokenizer_call_args
52        self._normalize = normalize
53
54    def _get_tokenizer_inputs(self, texts: "Sequence[str]") -> list[str]:
55        """Prepare input texts for tokenization.
56
57        :param texts: The texts to encode.
58        :return: The tokenizer inputs.
59        """
60        return list(texts)
61
62    def _aggregate_model_outputs(
63        self,
64        model_outputs: "BaseModelOutput",
65        model_inputs: "BatchEncoding",  # noqa: ARG002
66    ) -> torch.Tensor:
67        """Aggregate Transformer outputs using the CLS token (last hidden state).
68
69        Encoders overriding this function may make use of `model_inputs`.
70
71        :param model_outputs: The Transformer outputs.
72        :param model_inputs: The Transformer inputs (unused).
73        :return: The CLS token representations from the last hidden state.
74        """
75        return model_outputs.last_hidden_state[:, 0]
76
77    def _encode(self, texts: "Sequence[str]") -> "np.ndarray":
78        model_inputs = self._tokenizer(
79            self._get_tokenizer_inputs(texts),
80            return_tensors="pt",
81            **self._tokenizer_call_args,
82        ).to(self._device)
83
84        with torch.no_grad():
85            model_outputs = self._model(**model_inputs)
86            result = self._aggregate_model_outputs(model_outputs, model_inputs)
87            if self._normalize:
88                result = torch.nn.functional.normalize(result, p=2, dim=1)
89        return result.cpu().detach().numpy()

136class TCTColBERTDocumentEncoder(TransformerEncoder):
137    """Pre-trained TCT-ColBERT document encoder.
138
139    Adapted from Pyserini:
140    https://github.com/castorini/pyserini/blob/310c828211bb3b9528cfd59695184c80825684a2/pyserini/encode/_tct_colbert.py#L27
141
142    Corresponding paper: https://aclanthology.org/2021.repl4nlp-1.17/
143    """
144
145    def __init__(
146        self,
147        model: "str | Path" = "castorini/tct_colbert-msmarco",
148        device: str = "cpu",
149        max_length: int = 512,
150    ) -> None:
151        """Create a TCT-ColBERT document encoder.
152
153        :param model: Pre-trained TCT-ColBERT model (name or path).
154        :param device: PyTorch device.
155        :param max_length: Maximum number of tokens per document.
156        """
157        self._max_length = max_length
158        super().__init__(
159            model,
160            device=device,
161            tokenizer_call_args={
162                "max_length": max_length,
163                "padding": True,
164                "truncation": True,
165                "add_special_tokens": False,
166            },
167        )
168
169    def _get_tokenizer_inputs(self, texts: "Sequence[str]") -> list[str]:
170        return ["[CLS] [D] " + d for d in texts]
171
172    def _aggregate_model_outputs(
173        self,
174        model_outputs: "BaseModelOutput",
175        model_inputs: "BatchEncoding",
176    ) -> torch.Tensor:
177        token_embeddings = model_outputs.last_hidden_state[:, 4:, :]
178        input_mask_expanded = (
179            model_inputs.attention_mask[:, 4:]
180            .unsqueeze(-1)
181            .expand(token_embeddings.size())
182            .float()
183        )
184        sum_embeddings = torch.sum(token_embeddings * input_mask_expanded, 1)
185        sum_mask = torch.clamp(input_mask_expanded.sum(1), min=1e-9)
186        return sum_embeddings / sum_mask

229    def _aggregate_model_outputs(
230        self,
231        model_outputs: "BaseModelOutput",
232        model_inputs: "BatchEncoding",
233    ) -> torch.Tensor:
234        token_embeddings = model_outputs[0].masked_fill(
235            ~model_inputs.attention_mask[..., None].bool(), 0.0
236        )
237        return (
238            token_embeddings.sum(dim=1)
239            / model_inputs.attention_mask.sum(dim=1)[..., None]
240        )
